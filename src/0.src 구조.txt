src/

│   ├── common/        # 버튼, 모달 등 공통 컴포넌트

├── hooks/           # 커스텀 훅 (재사용 가능한 로직)
├── pages/           # 메인 페이지 컴포넌트
├── services/        # Firebase 등 외부 서비스 관련 로직
└── utils/           # 날짜 포맷 등 유틸리티 함수
